### linux-image

# Build Depends
sudo apt-get install build-essential cpio xz-utils bison flex

KERNVER=$(cat .config | grep 'Linux/x86' | awk '{print $3}')
BUILD_DIR=build/linux-image

# Kernel Configuration
make clean
make x86_64_defconfig
make menuconfig

# Build Kernel
make -j$(nproc) bzImage

# Install Kernel
mkdir -p $BUILD_DIR/boot
export INSTALL_PATH=$BUILD_DIR/boot
make -j$(nproc) install

# Build Modules
make -j$(nproc) modules

# Install Modules
export INSTALL_MOD_PATH=$BUILD_DIR
make -j$(nproc) modules_install
